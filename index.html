        function calculateLinearRegression(dataPoints) {

            const n = dataPoints.length; let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;

            dataPoints.forEach(p => { sumX += p.x; sumY += p.y; sumXY += p.x * p.y; sumXX += p.x * p.x; });

            const denominator = (n * sumXX - sumX * sumX);

            if (denominator === 0) return { m: 0, c: sumY / n };

           

            const m = (n * sumXY - sumX * sumY) / denominator;

            const c = (sumY - m * sumX) / n;

            return { m, c };

        }

            function animateTraining() {

                document.getElementById('train-progress-text').innerText = `${Math.floor(progress * 100)}%`;

                document.getElementById('train-loss').innerText = currentLoss.toFixed(4);

                document.getElementById('train-remain').innerText = `${((1 - progress) * (duration/1000)).toFixed(1)}s`;

                if (progress < 1) {

            } catch(e) { alert("QR Code format is unrecognized."); }

        }

        function applyImportedData(data) {

            setMode(data.m === 't' ? 'training' : 'simulator');

            else return;

                if(yInput && rowData.y && data.m === 't') yInput.value = rowData.y;

            });

        }
